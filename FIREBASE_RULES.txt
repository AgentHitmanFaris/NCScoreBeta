rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if the user is an admin
    function isAdmin() {
      return request.auth != null && (
        (exists(/databases/$(database)/documents/users/$(request.auth.uid)) && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true) ||
        exists(/databases/$(database)/documents/admins/$(request.auth.uid))
      );
    }

    // Users collection
    match /users/{userId} {
      allow read, write: if (request.auth != null && request.auth.uid == userId) || isAdmin();
    }

    // Songs collection
    match /songs/{songId} {
      allow read: if true;
      allow write: if isAdmin();

      // Arrangements subcollection
      match /arrangements/{arrangementId} {
        allow read: if true;
        allow write: if isAdmin();

        // Secure Content (Premium PDFs) - NESTED per your web snippet
        // Path: /songs/{songId}/arrangements/{arrangementId}/secure/content
        match /secure/content {
          allow read: if request.auth != null;
          allow write: if isAdmin();
        }
      }
    }
    
    // Artists collection
    match /artists/{artistId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Lyrics collection
    match /lyrics/{lyricId} {
      allow read: if true;
      allow write: if isAdmin();
    }
    
    // Bug Reports collection
    match /bug_reports/{reportId} {
      allow create: if true;
      allow read, write: if isAdmin();
    }
    
    // Legacy admins collection
    match /admins/{userId} {
      allow read: if (request.auth != null && request.auth.uid == userId) || isAdmin();
      allow write: if isAdmin();
    }
  }
}
